function updateUser(
  id: number,
  updates: Partial<User>
): Readonly<User> | undefined {
  const user = users.find((u) => u.id === id);
  if (!user) {
    return undefined;
  }

  const updatedUser = { ...user, ...updates };
  return updatedUser;
}
